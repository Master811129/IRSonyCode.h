# arduino-IRsonycode.h
the IRsonycode is the KEYWORDS of hex files for sony televisions. 


the commands:
POWER  ->    0xa90

B0    ->   0x910

B1    ->   0x10

B2    ->   0x810

B3    ->   0x410 

B4    ->   0xc10

B5    ->   0x210

B6    ->   0xa10

B7    ->   0x610

B8    ->   0xe10

B9    ->   0x110

VOLUME_UP (also you can use "VOL_UP")->   0x490

VOLUME_DOWN (also you can use "VOL_DOWN")->  0Xc90

CHANNEL_UP ->   0x90

CHANNEL_DOWN ->  0x890 

PICTURE_UP ->   0x190

PICTURE_DOWN ->  0x990 

TVVIDEO ->   0xA50

CLEAR  ->    0x2d0

JUMP  ->    0xdd0

MTS  ->     0xe90

DISPLAY  ->   0x5d0

SLEEP  ->    0x5d0 

MUTE  ->    0x290

REWIND  ->   0xd9a

PLAY   ->   0x59a 

PAUSE ->    0x99a

STOP  ->    0x19a

RECORD  ->   0xb9a

FF   ->     0x39a

ENTER  ->   0xd10 

EJECT ->    0x69a

BIT   ->    12
